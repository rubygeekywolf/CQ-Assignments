//created database
	CREATE DATABASE OnlineBookStore 

//created tables
	CREATE TABLE Book
	(
		Bid INT IDENTITY,					
		Bname VARCHAR(30),
		AuthorId INT,	
		Pid INT,
		ISBN VARCHAR(50),
		Price FLOAT,
		Genre VARCHAR(30),
		Quantity INT,
		CreatedOn DATETIME CONSTRAINT DF_Book_CreatedOn DEFAULT CURRENT_TIMESTAMP, 
		CONSTRAINT PK_Book PRIMARY KEY(Bid),										
		CONSTRAINT CHK_Book_Quantity CHECK (Quantity>0),										
		CONSTRAINT UQ_Book_ISBN UNIQUE (ISBN)											
)
	CREATE TABLE Author
	(
		AuthorId INT,
		AuthorName VARCHAR(30),
		CreatedOn DATETIME CONSTRAINT DF_Publisher_CreatedOn DEFAULT CURRENT_TIMESTAMP,
		CONSTRAINT PK_Publisher PRIMARY KEY(AuthorId)	
	)
	CREATE TABLE Publisher
	(

		Pid INT,
		Pname VARCHAR(30),
		CreatedOn DATETIME CONSTRAINT DF_Publisher_CreatedOn1 DEFAULT CURRENT_TIMESTAMP,
		CONSTRAINT PK_Publisher1 PRIMARY KEY(Pid)	
	)
	CREATE TABLE Customer
	(

		CustId INT,
		CustName VARCHAR(30),
		Address1 VARCHAR(max),
		Address2 VARCHAR(max),
		Phone INT,
		CreatedOn DATETIME CONSTRAINT DF_Customer_CreatedOn DEFAULT CURRENT_TIMESTAMP,
		CONSTRAINT PK_Customer PRIMARY KEY(CustId),
		CONSTRAINT UQ_Customer_Phone UNIQUE (Phone)
	)
	CREATE TABLE OrderDetail
	(

		OrderId INT,
		CustId INT not null,
		Bid INT not null,
		OrderDate DATE,
		Quantity INT,
		Amount FLOAT,
		CreatedOn DATETIME CONSTRAINT DF_OrderDetail_CreatedOn DEFAULT CURRENT_TIMESTAMP,
		CONSTRAINT PK_OrderDetail PRIMARY KEY(OrderId),
		CONSTRAINT CHK_OrderDetail_Quantity CHECK (Quantity>0)
	)
	CREATE TABLE Payment
	(

		PayId INT,
		CustId INT not null,
		OrderId INT not null,
		CardNo INT,
		CreatedOn DATETIME CONSTRAINT DF_Payment_CreatedOn DEFAULT CURRENT_TIMESTAMP,
		CONSTRAINT PK_Payment PRIMARY KEY(PayId)
	)
